<script>
  // ===================== CONFIG: PUT YOUR PHOTOS HERE =====================
  // Tip: use .jpg/.png (not huge). Add as many as you want.
  const PS_PHOTOS = [
    { src: "{{ '/images/me_01.jpg' | relative_url }}", alt: "Diego ‚Äî photo 1" },
    { src: "{{ '/images/me_02.jpg' | relative_url }}", alt: "Diego ‚Äî photo 2" },
    { src: "{{ '/images/me_03.jpg' | relative_url }}", alt: "Diego ‚Äî photo 3" },
    { src: "{{ '/images/me_04.jpg' | relative_url }}", alt: "Diego ‚Äî photo 4" },
    { src: "{{ '/images/me_05.jpg' | relative_url }}", alt: "Diego ‚Äî photo 5" },
    { src: "{{ '/images/me_06.jpg' | relative_url }}", alt: "Diego ‚Äî photo 6" }
  ];
  // ===================== END CONFIG =====================

  (function(){
    if (!PS_PHOTOS || PS_PHOTOS.length < 2) return;

    const main  = document.getElementById("ps-main");
    const mini1 = document.getElementById("ps-mini-1");
    const mini2 = document.getElementById("ps-mini-2");
    const mini3 = document.getElementById("ps-mini-3");
    const dots  = Array.from(document.querySelectorAll("#photo-showcase .ps-dot"));

    // preload
    PS_PHOTOS.forEach(p => { const img = new Image(); img.src = p.src; });

    const prefersReduced = window.matchMedia && window.matchMedia("(prefers-reduced-motion: reduce)").matches;
    const FADE_MS = prefersReduced ? 0 : 450;     // swap moment
    const HOLD_MS = 10000;                        // 10 seconds

    let lastSet = new Set();

    function shuffle(arr){
      const a = arr.slice();
      for (let i = a.length - 1; i > 0; i--){
        const j = Math.floor(Math.random() * (i + 1));
        [a[i], a[j]] = [a[j], a[i]];
      }
      return a;
    }

    function pickDistinct(k){
      const indices = shuffle([...PS_PHOTOS.keys()]);
      const picks = [];
      for (const idx of indices){
        if (!lastSet.has(idx)){
          picks.push(idx);
          if (picks.length === k) break;
        }
      }
      // if not enough (small photo list), allow repeats but still distinct within the set
      if (picks.length < k){
        for (const idx of indices){
          if (!picks.includes(idx)){
            picks.push(idx);
            if (picks.length === k) break;
          }
        }
      }
      return picks;
    }

    function swapImage(el, photo){
      if (!el) return;
      if (!prefersReduced) el.classList.add("is-leaving");
      setTimeout(() => {
        el.src = photo.src;
        el.alt = photo.alt || "";
        if (!prefersReduced) el.classList.remove("is-leaving");
      }, FADE_MS);
    }

    function tick(){
      // choose 4 different photos: 1 main + 3 minis
      const picks = pickDistinct(Math.min(4, PS_PHOTOS.length));
      lastSet = new Set(picks);

      const p0 = PS_PHOTOS[picks[0]];
      const p1 = PS_PHOTOS[picks[1] ?? picks[0]];
      const p2 = PS_PHOTOS[picks[2] ?? picks[0]];
      const p3 = PS_PHOTOS[picks[3] ?? picks[0]];

      swapImage(main,  p0);
      swapImage(mini1, p1);
      swapImage(mini2, p2);
      swapImage(mini3, p3);

      // animate dots
      const on = Math.floor(Math.random() * dots.length);
      dots.forEach((d,i)=>d.classList.toggle("is-on", i === on));
    }

    // first paint (no blank)
    main.src = PS_PHOTOS[0].src;
    main.alt = PS_PHOTOS[0].alt || "";
    if (mini1) { mini1.src = PS_PHOTOS[1 % PS_PHOTOS.length].src; mini1.alt = PS_PHOTOS[1 % PS_PHOTOS.length].alt || ""; }
    if (mini2) { mini2.src = PS_PHOTOS[2 % PS_PHOTOS.length].src; mini2.alt = PS_PHOTOS[2 % PS_PHOTOS.length].alt || ""; }
    if (mini3) { mini3.src = PS_PHOTOS[3 % PS_PHOTOS.length].src; mini3.alt = PS_PHOTOS[3 % PS_PHOTOS.length].alt || ""; }
    if (dots[0]) dots[0].classList.add("is-on");

    // rotate
    setTimeout(tick, 800);
    setInterval(tick, HOLD_MS);
  })();
</script>
